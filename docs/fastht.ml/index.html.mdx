\node (k0) at (-2.5, 1) [draw, secretdatum] {$k$};

\node (decap) at (0, -1) [draw, operation]
    {$\operatorname{Decap}(\mathit{sk}, c)$};
  \coordinate (decap-c) at ($(decap.north) - (1/2,0)$);
